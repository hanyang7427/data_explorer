Dataset Overview¶

  • 数据为拍拍贷真实业务数据，包含328553个信用标信息。
  • 包含8个特性：借款金额，借款期限，初始评级，逾期率，借款利率，是否首标，借款类型，借款日期

bin_log_edges = np.arange(trans(100), trans(500000)+0.5, 0.5)
bin_edges = 10 ** bin_log_edges
bin_centers = 10 ** (bin_log_edges + 0.5/2)[:-1]

def my_errorbar(data,label,color):
    data = data[data['借款期限']==label]
    # pd.cut分箱
    data_bins = pd.cut(data['借款金额'], bin_edges, right = False, include_lowest = True)
    # # 计算每个分组的统计量(均值)
    y_means = data['借款利率'].groupby(data_bins).mean()
    y_std = data['借款利率'].groupby(data_bins).std()
    # 绘制汇总数据图
    plt.errorbar(x = bin_centers, y = y_means, color=color, linewidth=3, label=label);
