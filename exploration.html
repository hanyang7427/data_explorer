df_lc_clean = df_lc_clean.loc[:,['ListingId','借款金额','借款期限','借款利率','初始评级','历史正常还款期数','历史逾期还款期数']]

# log变换(10为底)
def trans(x, inverse = False):
    if not inverse: return np.log10(x)
    else: return 10 ** x
# 创建log变换后的数据
df_lc_clean['log_借款金额'] = df_lc_clean['借款金额'].apply(trans)
# 绘制log处理后数据的直方图
