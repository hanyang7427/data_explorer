bin_log_edges = np.arange(trans(100), trans(500000)+0.5, 0.5)
bin_edges = 10 ** bin_log_edges
bin_centers = 10 ** (bin_log_edges + 0.5/2)[:-1]

def my_errorbar(data,label,color):
    data = data[data['']==label]
    # pd.cut
    data_bins = pd.cut(data[''], bin_edges, right = False, include_lowest = True)
    # # ()
    y_means = data[''].groupby(data_bins).mean()
    y_std = data[''].groupby(data_bins).std()
    # # 
    plt.errorbar(x = bin_centers, y = y_means, color=color, label=label);
